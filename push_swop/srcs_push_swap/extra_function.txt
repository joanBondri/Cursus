void	print_stack(t_list *a, t_list *b)
{
	t_list	*stack;
	t_list	*stock;

	printf("|%10s|%10s|\n", "A", "B");
	stack = a;
	stock = b;
	while (stack || stock)
	{
		if (stack && stock)
			printf("|%10i|%10i|\n", *((int*)stack->content),
					*((int*)stock->content));
		else if (stack)
			printf("|%10i|%10s|\n", *((int*)stack->content), "");
		else if (stock)
			printf("|%10s|%10i|\n", "", *((int*)stock->content));
		if (stack)
			stack = stack->next;
		if (stock)
			stock = stock->next;
	}
}
